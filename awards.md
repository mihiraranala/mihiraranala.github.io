<script>
(function () {
  var filterButtons = document.querySelectorAll('.filter-btn');
  var years = document.querySelectorAll('.award-year');
  var toggleAllBtn = document.querySelector('.awards-toggle-all');

  filterButtons.forEach(function (btn) {
    btn.addEventListener('click', function () {
      filterButtons.forEach(function (b) { b.classList.remove('active'); });
      btn.classList.add('active');
      var filter = btn.getAttribute('data-filter');

      years.forEach(function (year) {
        var items = year.querySelectorAll('.award-item');
        var visibleCount = 0;
        items.forEach(function (item) {
          var match = filter === 'all' || item.getAttribute('data-category') === filter;
          item.style.display = match ? '' : 'none';
          if (match) visibleCount++;
        });
        year.style.display = visibleCount > 0 ? '' : 'none';
        year.open = filter === 'all' ? year.getAttribute('data-default-open') === 'true' : true;
      });

      if (toggleAllBtn) {
        toggleAllBtn.setAttribute('data-state', filter === 'all' ? 'expand' : 'collapse');
        toggleAllBtn.textContent = filter === 'all' ? 'Expand all' : 'Collapse all';
      }
    });
  });

  if (toggleAllBtn) {
    toggleAllBtn.addEventListener('click', function () {
      var expand = toggleAllBtn.getAttribute('data-state') === 'expand';
      years.forEach(function (year) { year.open = expand; });
      toggleAllBtn.setAttribute('data-state', expand ? 'collapse' : 'expand');
      toggleAllBtn.textContent = expand ? 'Collapse all' : 'Expand all';
    });
  }
})();
</script>
